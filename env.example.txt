# इस फाइल का नाम बदलकर ".env" कर दें (बिना .example के) और नीचे अपनी वैल्यू भरें
# .env फाइल को कभी भी GitHub या किसी सार्वजनिक जगह अपलोड न करें

# आपकी client_secret_....json फाइल में से ये दोनों वैल्यू कॉपी करें
GOOGLE_CLIENT_ID=YOUR_CLIENT_ID_HERE.apps.googleusercontent.com
GOOGLE_CLIENT_SECRET=YOUR_CLIENT_SECRET_HERE

# Google Cloud Console > Credentials में इस OAuth Client के "Authorized redirect URIs" में
# यही URL जोड़ें:
# - लोकल टेस्टिंग के लिए:  http://localhost:3000/auth/google/callback
# - Render पर डिप्लॉय के बाद: https://आपका-app-नाम.onrender.com/auth/google/callback
GOOGLE_REDIRECT_URI=http://localhost:3000/auth/google/callback

# कोई भी लंबी, रैंडम स्ट्रिंग यहां डालें (सेशन को सुरक्षित रखने के लिए)
SESSION_SECRET=change-this-to-a-long-random-string

# सर्वर किस पोर्ट पर चलेगा (Render खुद अपना PORT सेट कर देता है, इसे वहां मत बदलें)
PORT=3000

# फ्रंटएंड कहां host होगा
# - लोकल टेस्टिंग के लिए: http://localhost:3000
# - Render पर डिप्लॉय के बाद: https://आपका-app-नाम.onrender.com
FRONTEND_URL=http://localhost:3000
